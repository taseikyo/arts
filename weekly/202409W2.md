```go
func twoSum(nums []int, target int) []int {
    cache := make(map[int]int)
    for idx, val := range nums {
        if preIdx, ok := cache[target-val]; ok {
            return []int{preIdx, idx}
        }
        cache[val] = idx
    }

    return []int{}
}
```
